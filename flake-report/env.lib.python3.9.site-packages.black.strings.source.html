21 def sub_twice(regex: Pattern[str], replacement: str, original: str) -> str:
22 """Replace `regex` with `replacement` twice on `original`.
23  
24 This is used by string normalization to perform replaces on
25 overlapping matches.
26 """
27 return regex.sub(replacement, regex.sub(replacement, original))

39 def lines_with_leading_tabs_expanded(s: str) -> List[str]:
40 """
41 Splits string into lines and expands only leading tabs (following the normal
42 Python rules)
43 """
44 lines = []
45 for line in s.splitlines():
46 # Find the index of the first non-whitespace character after a string of
47 # whitespace that includes at least one tab
48 match = FIRST_NON_WHITESPACE_RE.match(line)
49 if match:
50 first_non_whitespace_idx = match.start(1)
51  
52 lines.append(line[:first_non_whitespace_idx].expandtabs() + line[first_non_whitespace_idx:])
53 else:
54 lines.append(line)
55 return lines

58 def fix_docstring(docstring: str, prefix: str) -> str:
59 # https://www.python.org/dev/peps/pep-0257/#handling-docstring-indentation
60 if not docstring:
61 return ""
62 lines = lines_with_leading_tabs_expanded(docstring)
63 # Determine minimum indentation (first line doesn't count):
64 indent = sys.maxsize
65 for line in lines[1:]:
66 stripped = line.lstrip()
67 if stripped:
68 indent = min(indent, len(line) - len(stripped))
69 # Remove indentation (first line is special):
70 trimmed = [lines[0].strip()]
71 if indent < sys.maxsize:
72 last_line_idx = len(lines) - 2
73 for i, line in enumerate(lines[1:]):
74 stripped_line = line[indent:].rstrip()
75 if stripped_line or i == last_line_idx:
76 trimmed.append(prefix + stripped_line)
77 else:
78 trimmed.append("")
79 return "\n".join(trimmed)

101 def assert_is_leaf_string(string: str) -> None:
102 """
103 Checks the pre-condition that @string has the format that you would expect
104 of `leaf.value` where `leaf` is some Leaf such that `leaf.type ==
105 token.STRING`. A more precise description of the pre-conditions that are
106 checked are listed below.
107  
108 Pre-conditions:
109 * @string starts with either ', ", <prefix>', or <prefix>" where
110 `set(<prefix>)` is some subset of `set(STRING_PREFIX_CHARS)`.
111 * @string ends with a quote character (' or ").
112  
113 Raises:
114 AssertionError(...) if the pre-conditions listed above are not
115 satisfied.
116 """
117 dquote_idx = string.find('"')
118 squote_idx = string.find("'")
119 if -1 in [dquote_idx, squote_idx]:
120 quote_idx = max(dquote_idx, squote_idx)
121 else:
122 quote_idx = min(squote_idx, dquote_idx)
123  
124 assert 0 <= quote_idx < len(string) - 1, f"{string!r} is missing a starting quote character (' or \")."
125 assert string[-1] in (
126 "'",
127 '"',
128 ), f"{string!r} is missing an ending quote character (' or \")."
129 assert set(string[:quote_idx]).issubset(

134 def normalize_string_prefix(s: str) -> str:
135 """Make all string prefixes lowercase."""
136 match = STRING_PREFIX_RE.match(s)
137 assert match is not None, f"failed to match string {s!r}"
138 orig_prefix = match.group(1)
139 new_prefix = orig_prefix.replace("F", "f").replace("B", "b").replace("U", "").replace("u", "")
140  
141 # Python syntax guarantees max 2 prefixes and that one of them is "r"
142 if len(new_prefix) == 2 and "r" != new_prefix[0].lower():
143 new_prefix = new_prefix[::-1]
144 return f"{new_prefix}{match.group(2)}"

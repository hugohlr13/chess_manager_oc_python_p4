199 # you can't generally find out what module a function belongs to if you
200 # have only its name. Because of this, here are some callback functions
201 # that recognize if a gioven function belongs to a specific module
202 def module_callbacks():
203 def is_in_coroutine_module(name):
204 return name.startswith("coroutine.")

206 def is_in_modules_module(name):

208 return True

241 def get_newest_version():
242 f = urlopen("http://www.lua.org/manual/")
243 r = re.compile(r'^<A HREF="(\d\.\d)/">(Lua )?\1</A>')
244 for line in f:
245 m = r.match(line.decode("iso-8859-1"))
246 if m is not None:
247 return m.groups()[0]
248  
249 def get_lua_functions(version):
250 f = urlopen("http://www.lua.org/manual/%s/" % version)
251 r = re.compile(r'^<A HREF="manual.html#pdf-(?!lua|LUA)([^:]+)">\1</A>')
252 functions = []
253 for line in f:
254 m = r.match(line.decode("iso-8859-1"))
255 if m is not None:
256 functions.append(m.groups()[0])
257 return functions

259 def get_function_module(name):
260 for mod, cb in module_callbacks().items():
261 if cb(name):
262 return mod
263 if "." in name:
264 return name.split(".")[0]
265 else:
266 return "basic"

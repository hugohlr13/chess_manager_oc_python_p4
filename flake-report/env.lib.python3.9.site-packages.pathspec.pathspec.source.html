83 @classmethod
84 def from_lines(
85 cls: Type[Self],
86 pattern_factory: Union[str, Callable[[AnyStr], Pattern]],
87 lines: Iterable[AnyStr],
88 ) -> Self:
89 """
90 Compiles the pattern lines.
91  
92 *pattern_factory* can be either the name of a registered pattern
93 factory (:class:`str`), or a :class:`~collections.abc.Callable` used
94 to compile patterns. It must accept an uncompiled pattern (:class:`str`)
95 and return the compiled pattern (:class:`.Pattern`).
96  
97 *lines* (:class:`~collections.abc.Iterable`) yields each uncompiled
98 pattern (:class:`str`). This simply has to yield each line so it can
99 be a :class:`io.TextIOBase` (e.g., from :func:`open` or
100 :class:`io.StringIO`) or the result from :meth:`str.splitlines`.
101  
102 Returns the :class:`PathSpec` instance.
103 """
104 if isinstance(pattern_factory, str):
105 pattern_factory = util.lookup_pattern(pattern_factory)
106  
107 if not callable(pattern_factory):
108 raise TypeError(f"pattern_factory:{pattern_factory!r} is not callable.")
109  
110 if not _is_iterable(lines):
111 raise TypeError(f"lines:{lines!r} is not an iterable.")
112  
113 patterns = [pattern_factory(line) for line in lines if line]
114 return cls(patterns)

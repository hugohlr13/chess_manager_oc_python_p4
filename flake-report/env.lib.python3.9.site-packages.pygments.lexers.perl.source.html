401 def analyse_text(text):
402 if shebang_matches(text, r"perl"):
403 return True
404  
405 result = 0
406  
407 if re.search(r"(?:my|our)\s+[$@%(]", text):
408 result += 0.9
409  
410 if ":=" in text:
411 # := is not valid Perl, but it appears in unicon, so we should
412 # become less confident if we think we found Perl with :=
413 result /= 2
414  
415 return result
